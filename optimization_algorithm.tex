\begin{algorithm}
\caption{Sensitivity-Independent DAE Optimization with Hermite Interpolation}
\begin{algorithmic}[1]
\State \textbf{Input:} Initial parameters $p$, Time span $[t_0, t_f]$, Target data $(t^*, y^*)$, Tolerance $\epsilon$
\State \textbf{Initialize:} $p \leftarrow p_{init}$

\While{$||\nabla_p L|| > \epsilon$ and $iter < MAX\_ITER$}
    \State \textbf{1. Forward Solve (Black-box Solver)}
    \State Solve DAE $\to$ segments $\{ (t^{(i)}, x^{(i)}, \dot{x}^{(i)}, z^{(i)}) \}_{i=1}^S$ and events $\{ \tau_j \}$
    
    \State \textbf{2. Padding (JIT Compatibility)}
    \State Pad segments to fixed shapes $T \in \mathbb{R}^{S \times M}, X \in \mathbb{R}^{S \times M \times N_x}, \dot{X} \in \mathbb{R}^{S \times M \times N_x}$
    \State Extract event times $\mathcal{T} = [\tau_1, \dots, \tau_E, t_f, \dots]$ (padded)
    
    \State \textbf{3. Prediction \& Loss (JAX JIT)}
    \If{Method is Hermite}
        \State Construct dynamic time grid $t_{grid}(\mathcal{T})$ based on padded event times
        \State \textbf{Stop Gradient} on values: $\bar{X} \leftarrow \text{stop\_grad}(X), \bar{\dot{X}} \leftarrow \text{stop\_grad}(\dot{X})$
        \State Interpolate trajectory $x(t)$ using Cubic Hermite Spline with nodes $(\bar{X}, \bar{\dot{X}})$ on grid $t_{grid}$
        \State $y_{pred} = h(x(t^*), p)$
    \Else
        \State Use Linear or Sigmoid Blending interpolation
    \EndIf
    \State Compute Loss $L = \frac{1}{N} \sum ||y_{pred} - y^*||^2$
    \State Compute gradients w.r.t padded inputs: $\nabla_X L, \nabla_{\dot{X}} L, \nabla_Z L, \nabla_{\mathcal{T}} L$ 
    
    \State \textbf{4. Gradient Correction (for Hermite)}
    \If{Method is Hermite}
        \Comment{Account for sensitivity of derivatives $\dot{x}$ to state $x$ and params $p$}
        \State Compute Vector-Jacobian Product (VJP) correction:
        \State $\delta_x \leftarrow (\frac{\partial f}{\partial x})^T \nabla_{\dot{X}} L$, \quad $\delta_z \leftarrow (\frac{\partial f}{\partial z})^T \nabla_{\dot{X}} L$, \quad $\delta_p \leftarrow (\frac{\partial f}{\partial p})^T \nabla_{\dot{X}} L$
        \State Update gradients: $\nabla_X L \leftarrow \nabla_X L + \delta_x$, $\quad \nabla_p L \leftarrow \nabla_p L + \delta_p$
    \EndIf
    
    \State \textbf{5. Backward Sweep (Hybrid Adjoint)}
    \State Initialize $\lambda(t_f) = 0$, $\nabla_p^{tot} = \delta_p$ (accumulated)
    \For{segment $i = S$ down to $1$}
        \State Run discrete adjoint recursion on segment points to update $\lambda$ and accumulate $\nabla_p^{tot}$
        \State Add forcing terms $\nabla_X L, \nabla_Z L$ at data points
        \If{Event boundary}
            \State Apply jump conditions to $\lambda$ using $\nabla_{\mathcal{T}} L$
        \EndIf
    \EndFor
    
    \State \textbf{6. Update Parameters}
    \State $p \leftarrow \text{Adam}(p, \nabla_p^{tot})$
\EndWhile

\end{algorithmic}
\end{algorithm}
